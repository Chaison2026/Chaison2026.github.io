#define se second

using ll=long long;
const ll M=5e5+0xf;
ll n,m,s,head[M<<1],tot=1,vis[M<<1],dis[M<<1];
pair<pair<ll,ll>,ll>edge[M<<1];
void Google_Chrome(ll u,ll v,ll w){
    edge[tot].fi.fi=v;
    edge[tot].fi.se=w;
    edge[tot].se=head[u];
    head[u]=tot++;
}
void Dijkstra(ll start){
    for(int i=1;i<(M<<1);++i)dis[i]=INT_MAX;
    dis[start]=0,vis[start]=1;
    ll s=1,u=start;
    while(s<n){
        for(ll i=head[u];i!=-1;i=edge[i].se){
            ll v=edge[i].fi.fi,w=edge[i].fi.se;
            if(vis[v])continue;
            if(dis[v]>dis[u]+w)
                dis[v]=dis[u]+w;
        }
        ll min_num=0x7f7f7f7f,jl=0;
        for(ll i=1;i<=n;++i){
            if(vis[i])continue;
            if(min_num>=dis[i])
                min_num=dis[i],jl=i;
        }
        vis[jl]=1;
        u=jl;
        s++;
    }
}
signed main(void) {
    ios::sync_with_stdio(false),cin.tie(nullptr),cout.tie(nullptr);
    memset(head,-1,sizeof head);
    cin>>n>>m>>s;
    for(ll i=1,u,v,w;i<=m;++i){
        cin>>u>>v>>w;
        Google_Chrome(u,v,w);
    }
    Dijkstra(s);
    for(ll i=1;i<=n;++i)
        cout<<dis[i]<<' ';
    exit(0);
}

        while (ch < '0' || ch > '9') { if (ch == '-') f = -1; ch = getchar(); }

    }

inline void st_build() {
    for (int i = 1; i <= n; i++) st[i][0] = a[i];
    int t = log2(n);
    for (int j = 1; j <= t; j++) 
        for (int i = 1; i <= n - (1 << j) + 1; i++) 
            st[i][j] = max(st[i][j - 1], st[i + (1 << (j - 1))][j - 1]); 
}
inline int st_query(int l, int r) {
    int k = log2(r - l + 1);
    return max(st[l][k], st[r - (1 << k) + 1][k]);
}

struct edge
{
    int to, dis, next;
};

bool vis[MaxN];
int n, m, s;
inline void add_edge( int u, int v, int d )
{
    cnt++;
    e[cnt].dis = d;
    e[cnt].to = v;
    e[cnt].next = head[u];
    head[u] = cnt;
}
struct node
{
    int dis;
    int pos;
    bool operator <( const node &x )const
    {
        return x.dis < dis;
    }
};
std::priority_queue<node> q;
inline void dijkstra()
{
    dis[s] = 0;
    q.push( ( node ){0, s} );
    while( !q.empty() )
    {
        node tmp = q.top();
        q.pop();
        int x = tmp.pos, d = tmp.dis;
        if( vis[x] )
            continue;
        vis[x] = 1;
        for( int i = head[x]; i; i = e[i].next )
        {
            int y = e[i].to;
            if( dis[y] > dis[x] + e[i].dis )
            {
                dis[y] = dis[x] + e[i].dis;
                if( !vis[y] )
                {
                    q.push( ( node ){dis[y], y} );
                }
            }
        }
    }
}

{
